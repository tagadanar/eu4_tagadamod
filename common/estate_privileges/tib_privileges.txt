

tib_reduce_inflation_privilege = {
	icon = privilege_give_money
	is_valid = {
	}
	can_select = {
		inflation = 5
		adm_power = 75
		NOT = { has_country_modifier = tib_reduce_inflation }
	}
	on_granted = {
		add_adm_power = -75
		add_country_modifier = {
			name = tib_reduce_inflation
			duration = 730
		}
	}
	on_revoked = {
	}
	on_invalid = {
	}
	penalties = {
	}
	benefits = {
	}
	cooldown_years = 2
	on_cooldown_expires = {
		country_event = { id = tib_estates_modifiers.1 }
	}
	ai_will_do = { 
		factor = 10
		modifier = {
			factor = 0
			NOT = { inflation = 8 }
		}
	}
}
